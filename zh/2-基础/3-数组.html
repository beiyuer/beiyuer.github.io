数组

类型 数组名[数组大小]; 是数组的定义语句, 第5节类型将介绍定义语句的一般语法

a[i] 是数组中第i+1个元素

数组初始化

#include <litan.h>

int main() {

    int a[3];

    a[0] = get_int();
    a[1] = get_int();
    a[2] = get_int();
    // a[3] does not exist

    put_string("the second element in array `a` is ");
    put_int(a[1]);
    warp();

    put_string("the sum of elements in the array is ");
    put_int(a[0] + a[1] + a[2]);
    warp();

    return 0;
}
int a[3] = { get_int(), get_int(), get_int() }

数组与循环

#include <litan.h>

int main() {

    int i;
    int a[100];

    for (i = 0; i < 100; i = i + 1) {
        a[i] = i + 1;
    }

    int sum;
    sum = 0;

    for (i = 0; i < 100; i = i + 1) {
        sum = sum + a[i];
    }

    put_string("The sum from 1 to 100 is ");
    put_int(sum);

    return 0;
}
#include <litan.h>

int main() {

    int i;
    int a[100];

    for (i = 0; i < 10; i = i + 1) {
        a[i] = i + 1;
    }

    int product;
    product = 1;

    for (i = 0; i < 10; i = i + 1) {
        product = product * a[i];
    }

    put_string("The product from 1 to 10 is ");
    put_int(product);

    return 0;
}
#include <litan.h>

int main() {

    int i;
    int a[30];

    a[0] = 1;
    a[1] = 1;

    for (i = 2; i < 30; i = i + 1) {
        a[i] = a[i - 1] + a[i - 2];
    }

    for (i = 0; i < 30; i = i + 1) {
        put_int(a[i]);
        warp();
    }
}

字符串

字符串, 实质是字符的数组, 以结束字符 \0 作为字符串终止的标志 

#include <litan.h>

int main() {

    string str1 = "some text";
    put_string(str1);
    warp();

    char str2[10] = "some text";
    put_string(str2);
    warp();

    char str3[10] = { 's', 'o', 'm', 'e', ' ', 't', 'e', 'x', 't', '\0' };
    put_string(str3);
    warp();

    return 0;
}
