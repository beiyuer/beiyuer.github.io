#include <litan.h>

typedef struct { int x; int y; } vector2;

vector2 new_vector2(int x, int y) {
    vector2 result;
    result.x = x;
    result.y = y;
    return result;
}

void put_vector2(vector2 v) {
    put_char('(');
    put_int(v.x);
    put_string(", ");
    put_int(v.y);
    put_char(')');
}

vector2 add(vector2 x, vector2 y) {
    return new_vector2(x.x+y.x, x.y+y.y);
}

int dot(vector2 x, vector2 y) {
    return x.x * y.x + x.y * y.y;
}


int main() {

    vector2 a1;
    vector2 a2;

    a1 = new_vector2(1, 2);
    a2 = new_vector2(3, 4);

    put_vector2(a1);
    put_vector2(add(a1, a2));

    return 0;
}
