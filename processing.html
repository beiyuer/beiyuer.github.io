Input Event

state = False

def setup():
    size(800, 600)
    frameRate(60)

def draw(): 
    if state:
        ellipse(mouseX, mouseY, 20, 20)
    else:
        rect(mouseX-10, mouseY-10, 20, 20)

def mouseClicked(): 
    global state
    state = not state
